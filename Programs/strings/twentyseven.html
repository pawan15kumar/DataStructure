            void displayAlignedText(char text[], int alignment) {
                int length = strlen(text), i;
            
                if (alignment == 1) {
                    printf("%s\n", text); 
                } else if (alignment == 2) {
                    int spaces = (80 - length) / 2; 
                for (i = 0; i < spaces; i++) {
                    printf(" ");
                }
                printf("%s\n", text);
                } else if (alignment == 3) {
                int spaces = 80 - length;
                for (i = 0; i < spaces; i++) {
                    printf(" ");
                }
                printf("%s\n", text);
                } else {
                printf("Invalid alignment option.\n");
                }
            }
