#include
            void creatematrix(int ver, int edge, int adj[][6], int arr[][2]) {
                for (int i = 0; i < edge; i++) {
                    int x = arr[i][0];
                    int y = arr[i][1];
                    adj[x][y] = 1;
                    adj[y][x] = 1; 
                }
            }
            
            void printmatrix(int ver, int adj[][6]) {
                for (int i = 1; i <= ver; i++) {
                    for (int j = 1; j <= ver; j++) {
                        printf("%d ", adj[i][j]);
                    }
                    printf("\n");
                }
            }
            
            int main() {
                int ver, edge;
                printf("Enter the number of vertices: ");
                scanf("%d", &ver);
                printf("Enter the number of edges: ");
                scanf("%d", &edge);
                int arr[edge][2];
                int adj[6][6] = {0}; 
                printf("Enter the edges:\n");
                for (int i = 0; i < edge; i++) {
                    scanf("%d %d", &arr[i][0], &arr[i][1]);
                }
                creatematrix(ver, edge, adj, arr);
                printmatrix(ver, adj);
            }
