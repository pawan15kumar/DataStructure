#define MAX_SIZE 100

int bst[MAX_SIZE];

void initializeBST() {
	int i;
    for ( i = 0; i < MAX_SIZE; i++) {
        bst[i] = -1;
    }
}

void insert(int value) {
    int index = 0;

    while (bst[index] != -1) {
        if (value <= bst[index]) {
            index = 2 * index + 1;
        } else {
            index = 2 * index + 2;
        }
    }

    bst[index] = value;
}

int search(int value) {
    int index = 0;

    while (bst[index] != -1) {
        if (value == bst[index]) {
            return 1;
        } else if (value < bst[index]) {
            index = 2 * index + 1;
        } else {
            index = 2 * index + 2;
        }
    }

    return 0;
}

void preorder(int index) {
    if (bst[index] != -1) {
        printf("%d ", bst[index]);
        preorder(2 * index + 1);
        preorder(2 * index + 2);
    }
}

void inorder(int index) {
    if (bst[index] != -1) {
        inorder(2 * index + 1);
        printf("%d ", bst[index]);
        inorder(2 * index + 2);
    }
}

void postorder(int index) {
    if (bst[index] != -1) {
        postorder(2 * index + 1);
        postorder(2 * index + 2);
        printf("%d ", bst[index]);
    }
}

int findMinimum(int index) {
    while (bst[index * 2 + 1] != -1) {
        index = index * 2 + 1;
    }

    return (bst[index] != -1) ? bst[index] : -1;
}

int findSuccessor(int index) {
    index = 2 * index + 2;

    while (bst[index * 2 + 1] != -1) {
        index = index * 2 + 1;
    }

    return (bst[index] != -1) ? index : -1;
}

int del(int value) {
    int index = 0;
    int parent = -1;

    // Find the node to delete
    while (bst[index] != -1 && bst[index] != value) {
        parent = index;
        if (value < bst[index]) {
            index = 2 * index + 1;
        } else {
            index = 2 * index + 2;
        }
    }

    // If the value is not found
    if (bst[index] == -1) {
        return 0;
    }

    // Case 1: Node with no child
    if (bst[index * 2 + 1] == -1 && bst[index * 2 + 2] == -1) {
        bst[index] = -1;
    }
    // Case 2: Node with one child
    else if (bst[index * 2 + 1] == -1) {
        bst[index] = bst[index * 2 + 2];
        bst[index * 2 + 2] = -1;
    } else if (bst[index * 2 + 2] == -1) {
        bst[index] = bst[index * 2 + 1];
        bst[index * 2 + 1] = -1;
    }
    // Case 3: Node with two children
    else {
        int successor = findSuccessor(index);
        bst[index] = bst[successor];
        // Recursively delete the successor
        del(bst[successor]);
    }

    return 1;
}

int main() {
    initializeBST();

    insert(8);
    insert(3);
    insert(10);
    insert(1);
    insert(6);
    insert(9);
    insert(12);

    printf("Inorder traversal: ");
    inorder(0);
    printf("\nPreorder traversal: ");
    preorder(0);
    printf("\nPostorder traversal: ");
    postorder(0);

    printf("\n\nSearch for 6: %s", (search(6) ? "Found" : "Not found"));
    printf("\nSearch for 7: %s", (search(7) ? "Found" : "Not found"));

    printf("\n\nMinimum element in the tree: %d\n", findMinimum(0));

    // Delete node with value 6
    del(6);

    printf("\n\nAfter deleting 6:\n");
    printf("Inorder traversal: ");
    inorder(0);

    return 0;
}
