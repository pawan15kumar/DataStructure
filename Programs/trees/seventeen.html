struct Node {
            int data;
            struct Node* left;
            struct Node* right;
        };
        struct Node* createnewNode(int data) {
            struct Node* node = (struct Node*)malloc(sizeof(struct Node));
            node->data = data;
            node->left = node->right = NULL;
            return node;
        }
        
        int main() {
            int data;
            struct Node* root = NULL;
            struct Node* newNode = NULL;
            struct Node* currentNode = NULL;
            printf("Enter the root node value: ");
            scanf("%d", &data);
            root = createnewNode(data);
            currentNode = root;
            int choice;
            do {
                printf("Add a child to the current node (1: left, 2: right, 0: no child): ");
                scanf("%d", &choice);
        
                if (choice == 0) {
                    if (currentNode != root) {
                        currentNode = root;
                    } else {
                        printf("Tree construction completed.\n");
                        break;
                    }
                } else if (choice == 1 || choice == 2) {
                    printf("Enter the child node value: ");
                    scanf("%d", &data);
                    newNode = createnewNode(data);
        
                    if (choice == 1) {
                        currentNode->left = newNode;
                    } else if (choice == 2) {
                        currentNode->right = newNode;
                    }
                    currentNode = newNode;
                } else {
                    printf("Invalid choice, please enter 1, 2, or 0.\n");
                }
            } while (1);
        
            return 0;
        }
