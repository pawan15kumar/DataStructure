            int binarySearch(int arr[], int l, int u, int key) {
                int mid = l+(u - l)/2;
                if (l > u)
                {
                return -1;
                }
                if (arr[mid] == key)
                {
                return mid;
                }
                else if (arr[mid] > key){
                u = mid-1;
                }
                else{
                l = mid+1;
                }
            
                return binarySearch(arr, l, u, key);
            }
