            int bubbleSort(char ar[][100], int size, int index) {
                if (index == size - 1) {
                    return 0;
                }
            
                int j;
            
                for (j = 0; j < size - index - 1; j++) {
                    if (strcmp(ar[j], ar[j + 1]) > 0) {
                        char temp[100];
                        strcpy(temp, ar[j]);
                        strcpy(ar[j], ar[j + 1]);
                        strcpy(ar[j + 1], temp);
                    }
                }
            
                return bubbleSort(ar, size, index + 1);
            }
            
            int main() {
                char ar[10][100];
                int i, step = 0;
