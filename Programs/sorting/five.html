        void insertion(int ar[], int size) {
            int i, j, temp;
            for (i = 1; i < size; i++) {
                temp = ar[i];
                j = i - 1;
                while (temp < ar[j] && j >= 0) {
                    ar[j + 1] = ar[j];
                    j--;
                }
                ar[j + 1] = temp;
            }
        }
        
        int main() {
            int ar[10], i;
        
            printf("Enter the arrays\n");
            for (i = 0; i < 10; i++) {
                printf("Enter the array at location %d: ", i + 1);
                scanf("%d", &ar[i]);
            }
        
            insertion(ar, 10);
        
            printf("Sorted Arrays are\n");
            for (i = 0; i < 10; i++) {
                printf("%d\n", ar[i]);
            }
        
            return 0;
        }
