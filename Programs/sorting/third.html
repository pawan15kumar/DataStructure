            void bubble(int ar[], int size) {
                int i, j, temp;
                for (i = 1; i < size; i++) {
                    for (j = 0; j < size - i; j++) {
                        if (ar[j] > ar[j + 1]) {
                            temp = ar[j];
                            ar[j] = ar[j + 1];
                            ar[j + 1] = temp;
                        }
                    }
                }
            }
            
            int main() {
                int ar[10], i;
                clock_t start, end;
                double cpu_time;
            
                printf("Enter the arrays\n");
                for (i = 0; i < 10; i++) {
                    printf("Enter the array at location %d: ", i + 1);
                    scanf("%d", &ar[i]);
                }
            
                start = clock();
                bubble(ar, 10);
                end = clock();
            
                printf("Sorted Arrays are\n");
                for (i = 0; i < 10; i++) {
                    printf("%d\n", ar[i]);
                }
            
                cpu_time = ((double) (end - start)) / CLOCKS_PER_SEC;
                printf("Time taken by bubble sort: %f seconds\n", cpu_time);
            
                return 0;
            }
