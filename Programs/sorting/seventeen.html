            void merge_sort(char *a[], int low, int high) {
                int mid;
                if (low < high) {
                    mid = (low + high) / 2;
                    merge_sort(a, low, mid);
                    merge_sort(a, mid + 1, high);
                    merge(a, low, mid, high);
              
                }
            }
            void merge(char *a[], int low, int mid, int high) {
                int h, i, j, k;
                char *temp[n];
                h = low;
                i = low;
                j = mid + 1;
                while ((h <= mid) && (j <= high)) {
                    if (strcmp(a[h], a[j]) <= 0) {
                        temp[i] = a[h];
                        h = h + 1;
                    } else {
                        temp[i] = a[j];
                        j = j + 1;
                    }
                    i = i + 1;
                }
            
                while (j <= high) {
                    temp[i] = a[j];
                    j = j + 1;
                    i = i + 1;
                }
                while (h <= mid) {
                    temp[i] = a[h];
                    i = i + 1;
                    h = h + 1;
                }
                for (k = low; k <= high; k++)
                    a[k] = temp[k];
            }
