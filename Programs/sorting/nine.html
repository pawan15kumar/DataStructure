            void swap(int arr[], int a, int b) {
                int temp = arr[a];
                arr[a] = arr[b];
                arr[b] = temp;
            }
            int partition(int arr[], int low, int high) {
                int pivot = arr[low];
                int left = low + 1;
                int right = high;
            
                while (1) {
                    while (left <= right && arr[left] <= pivot)
                        left++;
            
                    while (arr[right] >= pivot && right >= left)
                        right--;
            
                    if (right < left)
                        break;
            
                    swap(arr, left, right);
                }
            
                swap(arr, low, right);
                return right;
            }
            
            void quicksort(int arr[], int low, int high) {
                int stack[10];
                int top = -1, pivot;
            
                stack[++top] = low;
                stack[++top] = high;
            
                while (top >= 0) {
                    high = stack[top--];
                    low = stack[top--];
            
                    pivot = partition(arr, low, high);
            
                    if (pivot - 1 > low) {
                        stack[++top] = low;
                        stack[++top] = pivot - 1;
                    }
            
                    if (pivot + 1 < high) {
                        stack[++top] = pivot + 1;
                        stack[++top] = high;
                    }
                }
            }
