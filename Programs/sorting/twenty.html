            void quick_sort(char *a[], int low, int high) {
                int pivot;
                if (low < high) {
                    pivot = partition(a, low, high);
                    quick_sort(a, low, pivot - 1);
                    quick_sort(a, pivot + 1, high);
                }
            }
            
            int partition(char *a[], int low, int high) {
                char *pivot;
                int i, j;
                char *temp;
                pivot = a[high];
                i = low - 1;
                for (j = low; j < high; j++) {
                    if (strcmp(a[j], pivot) <= 0) {
                        i++;
                        temp = a[i];
                        a[i] = a[j];
                        a[j] = temp;
                    
                    }
                }
                temp = a[i + 1];
                a[i + 1] = a[high];
                a[high] = temp;
            
                return (i + 1);
            }
