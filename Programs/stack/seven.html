            int push1(int stack[], int val, int top, int top2)
            {
                if (top + 1 == top2)
                {
                    return -1;
                }
                top++;
                stack[top] = val;
                return top;
            }
            
            int push2(int stack[], int val, int top, int top2)
            {
                if (top + 1 == top2)
                {
                    return -1;
                }
                top2--;
                stack[top2] = val;
                return top2;
            }
            
            int pop1(int stack[], int top)
            {
                if (top == -1)
                {
                    return -2;
                }
                top--;
                return top;
            }
            int pop2(int stack[], int top, int size)
            {
                if (top == size)
                {
                    return -1;
                }
                top++;
                return top;
            }
            int peek(int stack[], int top, int size)
            {
                if (top == -1 || top == size)
                {
                    return -1;
                }
                return top;
            }
