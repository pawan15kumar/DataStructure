        int push(char stack[], int size, int val, int i){
            if (i == size-1)
            {
                return -1;
            }
            i++;
            stack[i] = val;
            return i;
            
        }
        int pop(char stack[], int i){
            if (i == -1)
            {
                return -2;
            }
            i--;
            return i;
        }
