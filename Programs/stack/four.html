            struct node
            {
                char str[25];
                struct node *next;
            };
            typedef struct node Node;
            
            int isEmpty(Node *top)
            {
                if (top == NULL)
                {
                    return 1;
                }
                return 0;
            }
            
            int isFull(Node *ptr)
            {
                if (ptr == NULL)
                {
                    return 1;
                }
                return 0;
            }
            
            char* push(Node **top, char val[])
            {
                Node *nn;
                nn = (Node *)malloc(sizeof(Node));
                if (isFull(nn))
                {
                    return NULL;
                }
                strcpy(nn->str, val);
                nn->next = *top;
                *top = nn;
                return &(nn->str);
            }
            
            int pop(Node **top)
            {
                Node *temp;
                if (isEmpty(*top))
                {
                    return 0;
                }
                temp = *top;
                (*top) = temp->next;
                free(temp);
                return 1;
            }
            
            char* peek(Node *top)
            {
                if (isEmpty(top))
                {
                    return NULL;
                }
            
                return &(top->str);
            }
