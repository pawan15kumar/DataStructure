            int rear = -1;
            
            int insert(int que[], int size, int front) {
                int item;
                if ((front == 0 && rear == size - 1) || (front == (rear + 1) % size)) {
                    printf("Queue is full\n");
                } else if (rear == -1 && front == -1) {
                    front = rear = 0;
                    printf("Enter the item\n");
                    scanf("%d", &item);
                    que[rear] = item;
                } else {
                    rear = (rear + 1) % size;
                    printf("Enter the item\n");
                    scanf("%d", &item);
                    que[rear] = item;
                }
                return front;
            }
            
            int del(int que[], int front, int size) {
                int temp;
                if (front == -1) {
                    printf("Queue is empty\n");
                } else if (front == rear) {
                    temp = que[front];
                    printf("Delete item is %d\n", temp);
                    front = rear = -1;
                } else {
                    temp = que[front];
                    printf("Delete item is %d\n", temp);
                    front = (front + 1) % size;
                }
                return front;
            }
            
            void peek(int que[]) {
                int item;
                if (rear == -1) {
                    printf("Queue is empty\n");
                } else {
                    item = que[rear];
                    printf("Top element in the queue is %d\n", item);
                }
            }
            
            void isempty(int front) {
                if (front == -1) {
                    printf("Queue is empty\n");
                } else {
                    printf("Queue is not empty\n");
                }
            }
            
            void isfull(int size) {
                int front;
                if ((rear + 1) % size == front) {
                    printf("Queue is full\n");
                } else {
                    printf("Queue is not full\n");
                }
            }
