            struct que {
                char str[10];
                int priority;
                struct que* next;
            };
            
            struct que* rear = NULL;
            struct que* front = NULL;
            int insert(int count) {
                struct que* newnode = (struct que*) malloc(sizeof(struct que));
                printf("Enter the string: ");
                scanf("%s", newnode->str);
                printf("Enter the priority: ");
                scanf("%d", &(newnode->priority));
                newnode->next = NULL;
            
                if (rear == NULL) {
                rear = front = newnode;
                } else {
                if (newnode->priority > rear->priority) {
                    newnode->next = front;
                    front = newnode;
                } else {
                    while (rear->next != NULL && newnode->priority <= rear->priority) {
                    rear = rear->next;
                    }
                    rear->next = newnode;
                }
                }
                count++;
                printf("String %s with priority %d is inserted.\n", newnode->str, newnode->priority);
            return count;
            }
            
            int  del(int count) {
                if (front == NULL) {
                printf("Queue is empty\n");
                } else {
                struct que* temp = front;
                printf("Deleted string in queue is %s with priority %d\n",temp->str,temp->priority);
                    front = front->next;
                free(temp);
                count--;
                }
                return count;
            }
            
            void peek() {
                if (rear == NULL) {
                printf("Queue is empty\n");
                } else {
                printf("Top element in the queue is %s with priority %d\n", rear->str, rear->priority);
                }
            }
            
            void isempty() {
                if (rear == NULL) {
                printf("Queue is empty\n");
                } else {
                printf("Queue is not empty\n");
                }
            }
            
            void isfull() {
                printf("Queue is not full\n");
            }
            
            int main() {
                int ch,count=0;
                while (1) {

                printf("6. Total nodes\n");
                printf("7. Exit\n");

                scanf("%d", &ch);
                switch (ch) {
