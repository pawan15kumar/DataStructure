            struct que {
                char str[10];
                struct que *next;
            };
            
            struct que *rear = NULL, *front = NULL, *newnode, *temp;
            
            int insert(int count) {
                newnode = (struct que*)malloc(sizeof(struct que));
                if (newnode == NULL) {
                    printf("Memory allocation failed\n");
                    exit(1);
                }
            
                printf("Enter the string\n");
                scanf("%s", newnode->str);
                newnode->next = NULL;
            
                if (rear == NULL) {
                    rear = front = newnode;
                } else {
                    rear->next = newnode;
                    rear = newnode;
                }
            
                count++;
                return count;
            }
            
            int del(int count) {
                if (front == NULL) {
                    printf("Queue is empty\n");
                } else {
                    temp = front;
                    printf("Deleted string in queue is %s\n", temp->str);
                    front = front->next;
                    free(temp);
                    count--;
                }
                return count;
            }
            
            void peek() {
                if (rear == NULL) {
                    printf("Queue is empty\n");
                } else {
                    printf("Top string in queue is %s\n", front->str);
                }
            }
            
            void isempty() {
                if (rear == NULL) {
                    printf("Queue is empty\n");
                } else {
                    printf("Queue is not empty\n");
                }
            }
            
            void isfull() {
                printf("Queue is not full\n");
            }
            
            int main() {
                int ch, count = 0;
                while (1) {
                    printf("\n1. Insert\n");
                    printf("2. Delete\n");
                    printf("3. Peek\n");
                    printf("4. Is empty\n");
                    printf("5. Is full\n");
                    printf("6. Total nodes\n");
                    printf("7. Exit\n");
                    printf("Enter your choice\n");
                    scanf("%d", &ch);
                    switch (ch) {
                        case 1:
                            count = insert(count);
                            break;
                        case 2:
                            count = del(count);
                            break;
                        case 3:
                            peek();
                            break;
                        case 4:
                            isempty();
                            break;
                        case 5:
                            isfull();
                            break;
                        case 6:
                            printf("Total nodes in queue is %d\n", count);
                            break;
                        case 7:
                            exit(0);
                        default:
                            printf("Enter a valid number\n");
                    }
                }
            }
