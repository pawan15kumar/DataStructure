        int rear = -1;
        
        int insertfront(char que[][10], int size, int front) {
            char str[10];
            if (front == 0 && rear == size - 1 || front == rear + 1) {
                printf("Queue is full\n");
            } else if (front == -1 && rear == -1) {
                front = rear = 0;
                printf("Enter the string\n");
                scanf("%s", str);
                strcpy(que[front], str);
            } else if (front == 0) {
                front = size - 1;
                printf("Enter the string \n");
                scanf("%s", str);
                strcpy(que[front], str);
            } else {
                printf("Enter the string\n");
                scanf("%s", str);
                front--;
                strcpy(que[front], str);
            }
            return front;
        }
        
        int insertrear(char que[][10], int size, int front) {
            char str[10];
            if (front == 0 && rear == size - 1 || front == rear + 1) {
                printf("Queue is full\n");
            } else if (front == -1 && rear == -1) {
                front = rear = 0;
                printf("Enter the string\n");
                scanf("%s", str);
                strcpy(que[rear], str);
            } else if (rear == size - 1) {
                rear = 0;
                printf("Enter the string\n");
                scanf("%s", str);
                strcpy(que[rear], str);
            } else {
                rear++;
                printf("Enter the string\n");
                scanf("%s", str);
                strcpy(que[rear], str);
            }
            return front;
        }
        
        int delfront(char que[][10], int size, int front) {
            if (front == -1 && rear == -1) {
                printf("queue is empty\n");
            } else if (front == rear) {
                printf("Deleted string is %s\n", que[front]);
                front = rear = -1;
            } else if (front == size - 1) {
                printf("Delete string is %s\n", que[front]);
                front = 0;
            } else {
                printf("Deleted string is %s\n", que[front]);
                front++;
            }
            return front;
        }
        
        int delrear(char que[][10], int size, int front) {
            if (front == -1 && rear == -1) {
                printf("Queue is empty\n");
            } else if (front == rear) {
                printf("Deleted string is %s\n", que[rear]);
                front = rear = -1;
            } else if (rear == 0) {
                printf("Deleted string is %s\n", que[rear]);
                rear = size - 1;
            } else {
                printf("Deleted string is %s\n", que[rear]);
                rear--;
            }
            return front;
        }
        
        void peek(char que[][10], int front) {
            if (front == -1 && rear == -1) {
                printf("Queue is empty\n");
            } else {
                printf("Top element in the queue is %s\n", que[rear]);
            }
        }
        
        void isempty(int front) {
            if (front == -1 && rear == -1) {
                printf("Queue is empty\n");
            } else {
                printf("Queue is not empty\n");
            }
        }
        
        void isfull(int front, int size) {
            if (front == 0 && rear == size - 1 || front == rear + 1) {
                printf("Queue is full\n");
            } else {
                printf("Queue is not full\n");
            }
        }
